\chapter{热力学与统计力学：从熵定义到热力学定律}
\label{ch:thermo}

\section{概述}

R6 用 $S = k_B\ln W$ 定义了熵，将微观量子态（R4）与宏观热力学量（温度、压力、热量）连接起来。热力学的四大定律和量子统计分布都是 R6 在不同条件下的推论。

关键洞察：热力学第二定律 $\Delta S \geq 0$ 不是独立的物理原则——它是 $S=k\ln W$ 加上宇宙低熵初始条件（"过去假设"）的\textbf{统计推论}。

\section{第零定律——温度的传递性}

\begin{lawbox}[Zeroth Law of Thermodynamics]
  若系统 $A$ 与 $B$ 热平衡（$T_A=T_B$），且 $B$ 与 $C$ 热平衡（$T_B=T_C$），则 $A$ 与 $C$ 热平衡（$T_A=T_C$）。

  \textbf{前提}：热平衡状态的传递性——温度概念存在的逻辑前提
\end{lawbox}

\textbf{为什么叫"第零"定律？} 这条定律在逻辑上先于已经命名的第一和第二定律，但在历史上是最后才被明确提出来的（Fowler, 1931）。它确立了温度是一个良定义的物理量——如果热平衡不是传递的，"温度"这个概念就没有意义，温度计也无法工作。

\section{第一定律——能量守恒的热力学形式}

\begin{lawbox}[First Law of Thermodynamics]
\[
dU = T\,dS - P\,dV + \mu\,dN
\]
\textbf{前提}：R2（Noether 定理 → 能量守恒）+ R6（$S = k\ln W$ → 温度和熵的定义）

\textbf{变量}：
\begin{itemize}
  \item $U$ — 内能（J）。系统所有微观自由度的总能量（动能 + 势能）
  \item $T$ — 热力学温度（K）。由 $1/T \equiv \partial S/\partial U$ 定义
  \item $S$ — 熵（J/K）。$S = k_B\ln W$，衡量系统微观状态数
  \item $P$ — 压力（Pa = N/m$^2$）。$P \equiv T(\partial S/\partial V)_{U,N}$
  \item $V$ — 体积（m$^3$）
  \item $\mu$ — 化学势（J）。增加一个粒子所需的自由能代价
  \item $N$ — 粒子数
\end{itemize}
\end{lawbox}

\begin{derivationbox}[推导]
  \textbf{第 1 步：能量守恒。}
  Noether 定理（R2 的推论）：孤立系统的总能量守恒。热力学形式：$\Delta U = Q - W$（内能变化 = 吸收的热量 - 对外做的功）。

  \textbf{第 2 步：用熵表达可逆热交换。}
  由 R6 中温度的定义 $1/T = \partial S/\partial U$，对于可逆过程：$dQ_{\text{rev}} = T\,dS$。

  \textbf{第 3 步：用力学量表达功。}
  简单流体（如气体）对外做的功：$dW = P\,dV$（压力 $\times$ 体积变化）。

  \textbf{第 4 步：加入粒子交换。}
  若系统可与环境交换粒子，每个粒子携带化学势 $\mu$ 的自由能：$+\mu\,dN$。

  整理得：$dU = T\,dS - P\,dV + \mu\,dN$。$U$ 是状态函数（$dU$ 是恰当微分），与过程路径无关。
\end{derivationbox}

\section{第三定律——绝对零度不可达}

\begin{lawbox}[Third Law of Thermodynamics]
  $T \to 0$ 时，完美晶体的熵趋于零：$S \to 0$。

  \textbf{前提}：R6（$S = k\ln W$）+ R4（量子基态的非简并性）
\end{lawbox}

\begin{derivationbox}[推导]
  R6：$S = k_B\ln W$。温度趋近绝对零度时，系统必须处于量子基态（R4：能量是量子化的，存在最低能级）。

  如果基态是\textbf{非简并}的（只有一个最低能态），则 $W=1$，$S = k_B\ln 1 = 0$。

  如果基态有 $g$ 重简并（如冰 Ih 中的残余熵），则 $S(T\to 0) = k_B\ln g > 0$。

  \textbf{关键}：如果没有量子力学（如果能量是连续的），$T\to 0$ 时 $W$ 不会趋近 1，$S$ 也不会趋近 0。第三定律是量子力学在热力学中的体现。
\end{derivationbox}

\section{理想气体状态方程}

\begin{lawbox}[Ideal Gas Law]
\[
PV = nRT = Nk_B T
\]
\textbf{前提}：R6（$S = k\ln W$）+ 稀薄气体极限（无相互作用）

\textbf{变量}：
\begin{itemize}
  \item $P$ — 压力（Pa）
  \item $V$ — 体积（m$^3$）
  \item $n$ — 物质的量（mol）
  \item $N$ — 分子数
  \item $R = N_A k_B = 8.314$ J/(mol·K) — 气体常数
  \item $k_B = 1.381\times 10^{-23}$ J/K — 玻尔兹曼常数。SI 定义常数（exact）
  \item $T$ — 温度（K）
\end{itemize}
\end{lawbox}

\textbf{物理意义}：$PV = Nk_B T$ 可以理解为"压力 × 体积 = 分子数 × 每个分子的平均热运动能量"。$k_B T$（约 $4.1\times 10^{-21}$ J 在室温）是每个分子平动自由度的典型热运动能量。

\section{量子统计分布}

\begin{lawbox}[Fermi-Dirac 分布]
\[
f_{\text{FD}}(E) = \frac{1}{\exp[(E-\mu)/k_B T] + 1}
\]
\textbf{前提}：R6（熵最大化）+ 泡利不相容（费米子每态最多一个）
\end{lawbox}

\begin{lawbox}[Bose-Einstein 分布]
\[
f_{\text{BE}}(E) = \frac{1}{\exp[(E-\mu)/k_B T] - 1}
\]
\textbf{前提}：R6（熵最大化）+ 玻色子无占用限制
\end{lawbox}

\textbf{变量}：
\begin{itemize}
  \item $f(E)$ — 能量为 $E$ 的单粒子态的平均占据数（无量纲，0 ≤ $f$ ≤ 1 对 FD，$f$ ≥ 0 对 BE）
  \item $\mu$ — 化学势（J）。在 FD 分布中对应费米能级 $E_F$
  \item $k_B T$ — 热能标度。分母中的 $\pm 1$ 是量子统计与经典 Maxwell-Boltzmann 统计的唯一区别
\end{itemize}

\textbf{FD 与 BE 的物理区别}：
\begin{itemize}
  \item 费米子（$+1$）：$T \to 0$ 时，所有 $E < \mu$ 的态被完全填满（$f=1$），$E > \mu$ 的态完全空（$f=0$）。电子气形成\textbf{费米海}，这是金属导电和简并压的根源——白矮星依靠电子简并压抵抗引力坍缩。
  \item 玻色子（$-1$）：$T \to 0$ 时，大量粒子可以凝聚到最低能态（$\mu$ 趋近基态能量时 $f$ 发散），形成\textbf{Bose-Einstein 凝聚（BEC）}。这解释了超流 He-4 和原子 BEC 的实现。
\end{itemize}
