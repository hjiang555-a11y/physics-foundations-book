\chapter{热力学与统计力学：从熵定义到热力学定律}
\label{ch:thermo}

\section{概述}
R6 用 $S = k_B \ln W$ 定义了熵，通过计数微观状态数将量子力学（R4）与宏观热力学连接起来。热力学的四大定律都是 R6 在不同条件下的推论。

\section{第零定律：温度概念}
\begin{lawbox}[Zeroth Law]
  若 $T_A = T_B$ 且 $T_B = T_C$，则 $T_A = T_C$。
  \textbf{前提}：热平衡的传递性——温度的定义基础
\end{lawbox}
第零定律使得温度计成为可能。从统计力学角度看，热平衡意味着两个系统通过能量交换使总微观态数最大化。

\section{第一定律：能量守恒的热力学形式}

\begin{lawbox}[First Law of Thermodynamics]
\[
dU = T\,dS - P\,dV + \mu\,dN
\]
\textbf{前提}：R2（Noether $\to$ 能量守恒）+ R6（$S = k\ln W$）
\end{lawbox}

\begin{derivationbox}[推导]
  能量守恒（Noether）：$dU = dQ - dW$。
  可逆过程：$dQ_{\text{rev}} = T\,dS$（由 $1/T \equiv \partial S/\partial E$ 定义）。
  流体做功：$dW = P\,dV$。
  代入：$dU = T\,dS - P\,dV$。含粒子交换：$+ \mu\,dN$。
\end{derivationbox}

\section{第三定律：绝对零度不可达}

\begin{lawbox}[Third Law]
  $T \to 0$ 时，$S \to 0$（非简并基态）。
  \textbf{前提}：R6 + R4（量子基态结构）
\end{lawbox}

\begin{derivationbox}[推导]
  R6：$S = k_B \ln W$。$T\to 0$ 时系统占据量子基态。
  R4：若基态非简并，则 $W=1 \implies S=0$。
  若基态有 $g$ 重简并，$S(T=0) = k_B \ln g > 0$（残余熵，如冰 Ih）。
  经典物理会给出 $S \to -\infty$——量子力学（分立能谱）救了热力学。
\end{derivationbox}

\section{量子统计分布}

\begin{lawbox}[Fermi-Dirac Distribution]
\[
f_{\text{FD}}(E) = \frac{1}{\exp[(E-\mu)/k_B T] + 1}
\]
\textbf{前提}：R6 + R4（泡利 $\to$ 每态最多 1 个费米子）
\end{lawbox}

\begin{lawbox}[Bose-Einstein Distribution]
\[
f_{\text{BE}}(E) = \frac{1}{\exp[(E-\mu)/k_B T] - 1}
\]
\textbf{前提}：R6 + R4（叠加 $\to$ 玻色子无占用限制）
\end{lawbox}

分母中的 $\pm 1$ 是量子统计与经典 Maxwell-Boltzmann 统计的唯一区别——它决定了金属的导电性、半导体的能带结构、白矮星的稳定性、超流和 BEC 凝聚。
