\chapter{电磁学：从规范不变性到 Maxwell 方程组}
\label{ch:electromagnetism}

\section{概述}
从 R3（局域 $U(1)$ 规范不变性）加 R2（最小作用量），可以一条线推出全部电磁学。

\[
\text{R3 ($U(1)$规范)} + \text{R2 ($\delta S=0$)} \;\longrightarrow\; \text{Maxwell 四方程} \;\longrightarrow\; \text{电磁波}
\]

核心洞察：\textbf{Maxwell 方程组不是四个独立的经验定律，而是 $U(1)$ 规范理论的两个数学恒等式加两个 Euler-Lagrange 方程。}

\subsection{Ampère-Maxwell 定律}

\begin{lawbox}[Ampère-Maxwell Law]
\[
\nabla\times\mathbf{B} = \mu_0\mathbf{J} + \mu_0\varepsilon_0\frac{\partial\mathbf{E}}{\partial t}
\]
\textbf{前提}：$\partial_\mu F^{\mu\nu}=J^\nu$（Euler-Lagrange，需 R2）
\end{lawbox}

位移电流项 $\mu_0\varepsilon_0\partial\mathbf{E}/\partial t$ 是 Maxwell 为自洽性引入的理论修正——没有它，电荷守恒 $\nabla\cdot\mathbf{J}+\partial\rho/\partial t=0$ 与 Ampère 原定律矛盾。

\section{Lorentz 力定律}

\begin{lawbox}[Lorentz Force]
\[
\mathbf{F} = q(\mathbf{E} + \mathbf{v}\times\mathbf{B})
\]
\textbf{前提}：规范耦合的最小耦合原理 $p_\mu \to p_\mu - qA_\mu$
\end{lawbox}

\section{电磁波方程}

\begin{lawbox}[Electromagnetic Wave Equation]
\[
\nabla^2\mathbf{E} = \frac{1}{c^2}\frac{\partial^2\mathbf{E}}{\partial t^2},\quad
\nabla^2\mathbf{B} = \frac{1}{c^2}\frac{\partial^2\mathbf{B}}{\partial t^2},\quad
c = \frac{1}{\sqrt{\varepsilon_0\mu_0}}
\]
\textbf{前提}：Maxwell 方程组真空解（$\rho=0,\mathbf{J}=0$）
\end{lawbox}

\begin{derivationbox}[推导]
  真空 Maxwell：$\nabla\times\mathbf{E}=-\partial_t\mathbf{B}$，$\nabla\times\mathbf{B}=\mu_0\varepsilon_0\partial_t\mathbf{E}$。
  取旋度后者：$\nabla\times(\nabla\times\mathbf{B})=\mu_0\varepsilon_0\nabla\times\partial_t\mathbf{E}$。
   用 $\nabla\times(\nabla\times\mathbf{B})=\nabla(\underbrace{\nabla\cdot\mathbf{B}}_{=0})-\nabla^2\mathbf{B}$ 和 Faraday：$\nabla\times\partial_t\mathbf{E}=-\partial_t^2\mathbf{B}$。
  得 $\nabla^2\mathbf{B}=\mu_0\varepsilon_0\partial_t^2\mathbf{B}$。令 $c=1/\sqrt{\mu_0\varepsilon_0}$，即得波动方程。
\end{derivationbox}
