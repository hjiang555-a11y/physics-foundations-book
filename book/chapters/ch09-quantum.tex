\chapter{量子力学：从量子公设到薛定谔方程}
\label{ch:quantum}

\section{概述}
R4（四大量子公设）加 R5（Born 规则）定义了量子力学的全部计算规则。本章从这些公设逐一推出现代量子力学的核心方程式。

\section{薛定谔方程}

\begin{lawbox}[Schrödinger Equation]
\[
i\hbar\frac{\partial\psi}{\partial t} = -\frac{\hbar^2}{2m}\nabla^2\psi + V\psi
\]
\textbf{前提}：R4(4b) 幺正演化 + R4(4c) 正则对易 + R4(4d) 算符对应
\end{lawbox}

\begin{derivationbox}[推导：态向量 → 坐标表象]
  R4(4b)：$i\hbar\frac{d}{dt}|\psi\rangle = \hat{H}|\psi\rangle$。
  R4(4c)：$[\hat{x},\hat{p}]=i\hbar \implies \hat{p} = -i\hbar\nabla$（坐标表象）。
  R4(4d)：$\hat{H} = \frac{\hat{p}^2}{2m} + V(\hat{x})$。
  投影到坐标基底 $|\mathbf{r}\rangle$：
  \[
  i\hbar\frac{\partial}{\partial t}\langle\mathbf{r}|\psi\rangle = \left[-\frac{\hbar^2}{2m}\nabla^2 + V(\mathbf{r})\right]\langle\mathbf{r}|\psi\rangle
  \]
  即 $i\hbar\partial_t\psi = \hat{H}\psi$。
\end{derivationbox}

\section{$E = h\nu$（普朗克-爱因斯坦关系）}

\begin{lawbox}[Planck-Einstein Relation]
\[
E = h\nu = \hbar\omega
\]
\textbf{前提}：R4(4c) → 谐振子量子化
\end{lawbox}

\begin{derivationbox}[谐振子量子化]
  经典谐振子 $H = \frac{p^2}{2m} + \frac12 m\omega^2 x^2$。
  算符化：定义产生/湮灭算符 $a^\dagger, a$（满足 $[a,a^\dagger]=1$）：
  \[
  \hat{H} = \hbar\omega\!\left(a^\dagger a + \frac12\right)
  \]
  本征值：$E_n = \hbar\omega(n+1/2) = h\nu(n+1/2)$。
  相邻能级差 $h\nu$。光子就是 $\nu$ 频率的电磁场量子——$E=h\nu$。
\end{derivationbox}

\section{de Broglie 关系}

\begin{lawbox}[de Broglie Wavelength]
\[
\lambda = \frac{h}{p}
\]
\textbf{前提}：$E=h\nu$ + 光子动量 $p=E/c$
\end{lawbox}

光子：$E = h\nu = hc/\lambda$，$p = E/c \implies \lambda = h/p$。
de Broglie (1924) 大胆推广：\textbf{所有物质都有波动性}。Davisson-Germer (1927) 电子衍射实验证实。

\section{不确定性原理}

\begin{lawbox}[Heisenberg Uncertainty Principle]
\[
\sigma_x \sigma_p \geq \frac{\hbar}{2}, \qquad \Delta E\,\Delta t \geq \frac{\hbar}{2}
\]
\textbf{前提}：$[x,p]=i\hbar$（R4(4c)）+ 一般不等式
\end{lawbox}

\begin{derivationbox}[Kennard 不等式]
  对任意 $A,B$：$\sigma_A\sigma_B \geq \frac12|\langle[A,B]\rangle|$。
  取 $A=x, B=p$，代入 $[x,p]=i\hbar$ 得 $\sigma_x\sigma_p \geq \hbar/2$。
  最小值在高斯波包：$\psi(x) \propto e^{-x^2/4\sigma_x^2}$。
\end{derivationbox}

\section{泡利不相容原理}

\begin{lawbox}[Pauli Exclusion Principle]
  两个全同费米子不能占据完全相同的量子态。
  \textbf{前提}：自旋-统计定理（R1 + R4）
\end{lawbox}

\textbf{为什么原子不坍缩？} 泡利原理阻止所有电子落到最低能级——迫使电子占据越来越高的能级，产生元素周期表的壳层结构。\textbf{这是物质稳定性的微观根源。}
