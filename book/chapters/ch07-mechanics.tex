\chapter{力学：从最小作用量到万有引力}
\label{ch:mechanics}

\section{概述}

力学是 R2（最小作用量原理）最直接的产物。给定系统的拉格朗日量 $L = T - V$，Euler-Lagrange 方程自动产生全部运动方程。本章展示从 $\delta S = 0$ 出发，依次推出 Euler-Lagrange 方程、Noether 定理、Newton 三大定律、万有引力定律和 Kepler 第三定律。

每条定律标注逻辑前提、推导步骤、必要条件 \texttt{[N]} 和充分条件 \texttt{[S]}。

\section{Euler-Lagrange 方程}
% ════════════════════════════════════════════════════════

\begin{lawbox}[Euler-Lagrange 方程]
  \[
  \frac{d}{dt}\!\left( \frac{\partial L}{\partial \dot{q}_i} \right) - \frac{\partial L}{\partial q_i} = 0
  \]
  \textbf{前提}：R2（$\delta S = 0$）\quad \textbf{推导号}：\texttt{deriv.euler\_lagrange}
\end{lawbox}

\subsection{推导}

\begin{derivationbox}[推导：从最小作用量到 Euler-Lagrange 方程]
  \textbf{Step 1} — 作用量定义：
  $S[q] = \int_{t_1}^{t_2} L(q,\dot{q},t)\,dt$
  \hfill\textit{(R2)}

  \textbf{Step 2} — 路径变分，端点固定：
  $\delta S = \int [L(q+\delta q, \dot{q}+\delta\dot{q},t) - L(q,\dot{q},t)]\,dt$
  \hfill\texttt{[N]} $\delta q(t_1)=\delta q(t_2)=0$

  \textbf{Step 3} — Taylor 展开至一阶：
  $\delta S = \sum_i \int [\frac{\partial L}{\partial q_i}\delta q_i + \frac{\partial L}{\partial \dot{q}_i}\delta\dot{q}_i]\,dt$
  \hfill\texttt{[N]} $L \in C^2$

  \textbf{Step 4} — 分部积分第二项：
  $\int \frac{\partial L}{\partial\dot{q}_i}\frac{d}{dt}(\delta q_i)\,dt = [\frac{\partial L}{\partial\dot{q}_i}\delta q_i]_{t_1}^{t_2} - \int \frac{d}{dt}(\frac{\partial L}{\partial\dot{q}_i})\delta q_i\,dt$

  \textbf{Step 5} — 边界项因固定端点消失 $\to$ 整理：
  $\delta S = \sum_i \int [\frac{\partial L}{\partial q_i} - \frac{d}{dt}(\frac{\partial L}{\partial\dot{q}_i})]\delta q_i\,dt$

  \textbf{Step 6} — $\delta S = 0$ (R2) + 变分学基本引理（$\delta q_i$ 任意）：
  $\frac{\partial L}{\partial q_i} - \frac{d}{dt}(\frac{\partial L}{\partial\dot{q}_i}) = 0$
  \hfill\texttt{[N]} $\delta q_i$ 任意且独立
\end{derivationbox}

\textbf{必要条件}：R2 ($\delta S=0$), $L \in C^2$, 固定端点, $\delta q_i$ 任意独立。

\textbf{充分条件}：给定 $C^2$ 拉格朗日量和固定端点，$\delta S = 0 \iff$ Euler-Lagrange 方程。\texttt{reversible: true}

\section{Noether 定理}
% ════════════════════════════════════════════════════════

\begin{lawbox}[Noether 定理]
  作用量 $S$ 的每个连续对称性变换对应一个守恒的 Noether 荷 $Q$：
  \[
  Q \equiv \sum_i \frac{\partial L}{\partial \dot{q}_i}\,Q_i - F, \qquad \frac{dQ}{dt} = 0
  \]
  \textbf{前提}：R2, R1\quad \textbf{推导号}：\texttt{deriv.noether\_theorem}
\end{lawbox}

\subsection{推导}

\begin{derivationbox}[推导：Noether 定理]
  \textbf{Step 1} — 对称性变换：$q_i \to q_i' = q_i + \varepsilon Q_i(q,\dot{q},t)$
  \hfill\texttt{[N]} $Q_i$ 可微，$\varepsilon$ 为 Lie 群参数

  \textbf{Step 2} — 对称性条件：$S[q'] = S[q] + \varepsilon[F(t_2)-F(t_1)]$
  $\implies L' = L + \varepsilon\,dF/dt$
  \hfill\texttt{[N]} 作用量必须不变（至多差边界项）

  \textbf{Step 3} — 展开 $L'$：$L' - L = \varepsilon\sum_i[\frac{\partial L}{\partial q_i}Q_i + \frac{\partial L}{\partial\dot{q}_i}\dot{Q}_i]$

  \textbf{Step 4} — 用 Euler-Lagrange 替换 $\partial L/\partial q_i$：
  $\frac{\partial L}{\partial q_i} = \frac{d}{dt}(\frac{\partial L}{\partial\dot{q}_i})$
  \hfill\texttt{[N]} 系统沿真实路径（on-shell）

  \textbf{Step 5} — 重组为全导数：
  $\sum_i[\frac{d}{dt}(\frac{\partial L}{\partial\dot{q}_i})Q_i + \frac{\partial L}{\partial\dot{q}_i}\dot{Q}_i] = \frac{d}{dt}(\sum_i \frac{\partial L}{\partial\dot{q}_i}Q_i)$

  \textbf{Step 6} — 与对称性条件联立：
  $\frac{d}{dt}(\sum_i \frac{\partial L}{\partial\dot{q}_i}Q_i - F) = 0$

  \textbf{Step 7} — 定义守恒荷：$Q \equiv \sum_i \frac{\partial L}{\partial\dot{q}_i}Q_i - F$，$dQ/dt = 0$
\end{derivationbox}

\textbf{必要条件}：R2, R1（时间坐标）, 连续对称性（Lie 群）, $L \in C^2$, Euler-Lagrange 成立（on-shell）。

\textbf{充分条件}：连续对称性 + Euler-Lagrange $\Rightarrow$ 守恒流。\texttt{reversible: true}（逆 Noether 定理：每个守恒量对应一对称性）。

\section{Newton 运动定律}
% ════════════════════════════════════════════════════════

\subsection{第一定律（惯性定律）}

\begin{lawbox}[Newton 第一定律]
  $F = 0 \implies d\mathbf{v}/dt = 0$（匀速直线运动）
  \quad\textbf{前提}：R2, R1\quad\textbf{推导号}：\texttt{deriv.newton\_first}
\end{lawbox}

\noindent\textbf{变量}：力 $F$ [M$\cdot$L$\cdot$T$^{-2}$], 速度 $\mathbf{v}$ [L$\cdot$T$^{-1}$], 质量 $m$ [M]（常数）。

\begin{derivationbox}[推导]
  \textbf{Step 1} — 自由粒子 Lagrangian：$L = \frac{1}{2}m|\mathbf{v}|^2$（$V$ = 常数 $\implies \nabla V = 0$）\\
  \textbf{Step 2} — Euler-Lagrange：$d(m\mathbf{v})/dt = -\nabla V = 0$
  \texttt{[N]} $m$ 常数\quad\texttt{[N]} 惯性系\\
  \textbf{Step 3} — $\mathbf{v} = $ 常数。\quad\texttt{[S]} 自由粒子 $L$ + Euler-Lagrange $\Rightarrow$ 第一定律。
\end{derivationbox}

\subsection{第二定律（运动方程）}

\begin{lawbox}[Newton 第二定律]
  $\mathbf{F} = d\mathbf{p}/dt$（一般形式）；$m$ 常数时 $\mathbf{F} = m\mathbf{a}$
  \quad\textbf{前提}：R2, Euler-Lagrange, R1
\end{lawbox}

\noindent\textbf{变量}：动量 $\mathbf{p} = m\mathbf{v}$ [M$\cdot$L$\cdot$T$^{-1}$], 加速度 $\mathbf{a} = d\mathbf{v}/dt$ [L$\cdot$T$^{-2}$],
力 $\mathbf{F} \equiv -\nabla V(\mathbf{r})$ [M$\cdot$L$\cdot$T$^{-2}$]。

\begin{derivationbox}[推导]
  \textbf{Step 1} — $L = \frac{1}{2}m|\mathbf{v}|^2 - V(\mathbf{r})$\quad\texttt{[N]} 保守力（$V$ 存在）\\
  \textbf{Step 2} — Euler-Lagrange：$d(m\mathbf{v})/dt = -\nabla V$\\
  \textbf{Step 3} — 定义 $\mathbf{F} \equiv -\nabla V \implies d\mathbf{p}/dt = \mathbf{F}$
  \quad\texttt{[S]} $L = T-V$ + Euler-Lagrange $\Rightarrow$ 第二定律。
\end{derivationbox}

\subsection{第三定律（作用-反作用）}

\begin{lawbox}[Newton 第三定律]
  $\mathbf{F}_{12} = -\mathbf{F}_{21}$（弱形式：大小相等方向相反；强形式 collinear 在 EM 中破缺）
  \quad\textbf{前提}：Noether 定理, R1
\end{lawbox}

\begin{derivationbox}[推导]
  \textbf{Step 1} — 孤立系统动量守恒（Noether）：$d(\mathbf{p}_1 + \mathbf{p}_2)/dt = 0$
  \texttt{[N]} 孤立系统（无外力）\\
  \textbf{Step 2} — Newton 第二定律：$d\mathbf{p}_1/dt = \mathbf{F}_{12}$，$d\mathbf{p}_2/dt = \mathbf{F}_{21}$\\
  \textbf{Step 3} — $\mathbf{F}_{12} + \mathbf{F}_{21} = 0 \implies \mathbf{F}_{12} = -\mathbf{F}_{21}$
  \quad\texttt{[S]} 动量守恒 + 第二定律 $\Rightarrow$ 第三定律弱形式。
\end{derivationbox}

\subsection{万有引力定律}

\begin{lawbox}[Newton 万有引力定律]
  $\mathbf{F}_{12} = -G\,\dfrac{m_1 m_2}{r^2}\,\hat{\mathbf{r}}$
  \quad\textbf{前提}：Einstein 场方程（弱场极限）, R1
\end{lawbox}

\noindent\textbf{变量}：引力常数 $G = 6.67430(15)\times10^{-11}$ m$^3\cdot$kg$^{-1}\cdot$s$^{-2}$ [M$^{-1}\cdot$L$^3\cdot$T$^{-2}$],
引力势 $\phi = -GM/r$ [L$^2\cdot$T$^{-2}$], 质量密度 $\rho$ [M$\cdot$L$^{-3}$]。

\begin{derivationbox}[推导：从 Einstein 场方程到 Newton 引力]
  \textbf{Step 1} — 弱场近似：$g_{\mu\nu} = \eta_{\mu\nu} + h_{\mu\nu}$，$|h_{\mu\nu}| \ll 1$
  \texttt{[N]} weak\_field\_limit\\
  \textbf{Step 2} — Einstein 场方程 (0,0) 分量弱场展开：$R_{00} \approx -\frac{1}{2}\nabla^2 h_{00}$\\
  \textbf{Step 3} — 从测地线方程得 $h_{00} = 2\phi/c^2$\\
  \textbf{Step 4} — 化为 Poisson 方程：$\nabla^2\phi = 4\pi G\rho$\\
  \textbf{Step 5} — 球对称解：$\phi = -GM/r$，$\mathbf{F} = -m\nabla\phi = -GmM/r^2\,\hat{\mathbf{r}}$
  \quad\texttt{[S]} Einstein 场方程 + 弱场低速极限 $\Rightarrow$ Newton 引力。
  \texttt{reversible: false}（Newton 理论是 GR 的低能有效理论）。
\end{derivationbox}

\section{Kepler 第三定律（推论）}
% ════════════════════════════════════════════════════════

\begin{lawbox}[Kepler 第三定律 — corollary]
  $T^2 \propto r^3$（轨道周期的平方正比于半长轴的立方）
  \quad\textbf{前提}：Newton 第二 + 万有引力, R1
\end{lawbox}

\noindent\textbf{变量}：轨道周期 $T$ [T], 圆轨道半径 $r$ [L]。

\begin{derivationbox}[推导]
  \textbf{Step 1} — 圆轨道向心加速度：$a = v^2/r$；Newton 第二：$F = mv^2/r$
  \texttt{[N]} 圆轨道近似（$e=0$）\\
  \textbf{Step 2} — 万有引力提供向心力：$mv^2/r = GMm/r^2 \implies v^2 = GM/r$\\
  \textbf{Step 3} — 轨道周期：$T = 2\pi r/v \implies T^2 = (4\pi^2/GM)\,r^3$
  \quad\texttt{[S]} Newton 力学 + 万有引力 $\Rightarrow$ Kepler 第三定律。
\end{derivationbox}
