\chapter{力学：从最小作用量到牛顿定律}
\label{ch:mechanics}

\section{概述}

力学是 R2（最小作用量原理）最直接的产物。给定一个系统的拉格朗日量 $L$（编码了系统的全部动力学信息），Euler-Lagrange 方程就能产生全部运动方程。

本章展示如何从 $\delta S = 0$ 出发，依次推出 Euler-Lagrange 方程、牛顿三大定律、万有引力定律和 Kepler 行星运动定律。每条定律都标注了逻辑前提和推导步骤。

\section{Euler-Lagrange 方程}

Euler-Lagrange 方程不是物理假设，而是变分法（calculus of variations）的数学定理。R2 断言真实运动使作用量取极值，Euler-Lagrange 方程就是这个条件的数学等价形式。

\begin{lawbox}[Euler-Lagrange 方程]
\[
\frac{d}{dt}\!\left( \frac{\partial L}{\partial \dot{q}_i} \right) - \frac{\partial L}{\partial q_i} = 0
\]
\textbf{前提}：R2（$\delta S = 0$），端点固定（$\delta q(t_1)=\delta q(t_2)=0$）

\textbf{变量说明}：
\begin{itemize}
  \item $L(q_i,\dot{q}_i,t)$ — 拉格朗日量（Lagrangian），标准形式为动能减势能：$L = T - V$
  \item $q_i$ — 第 $i$ 个广义坐标（generalized coordinate），描述系统位形的独立变量。笛卡尔坐标是特例，广义坐标也可是角度等
  \item $\dot{q}_i = dq_i/dt$ — 广义速度
  \item $\partial L/\partial q_i$ — 广义力（系统如何"倾向于"改变 $q_i$）
  \item $\partial L/\partial \dot{q}_i = p_i$ — 广义动量（与 $q_i$ 共轭的动量）
\end{itemize}
\end{lawbox}

\begin{derivationbox}[推导：从变分原理到 Euler-Lagrange 方程]
  \textbf{第 1 步：写出作用量变分。}
  \[
  S[q] = \int_{t_1}^{t_2} L(q,\dot{q},t)\,dt
  \]
  考虑路径的微小变分 $q(t) \to q(t)+\delta q(t)$，端点固定：$\delta q(t_1)=\delta q(t_2)=0$。

  \textbf{第 2 步：展开到一阶。}
  \[
  \delta S = \int_{t_1}^{t_2}\!\left[\frac{\partial L}{\partial q}\delta q + \frac{\partial L}{\partial \dot{q}}\delta\dot{q}\right]dt
  \]
  其中 $\delta\dot{q} = \frac{d}{dt}(\delta q)$。

  \textbf{第 3 步：分部积分消除 $\delta\dot{q}$。}
  \[
  \int_{t_1}^{t_2}\frac{\partial L}{\partial\dot{q}}\frac{d}{dt}(\delta q)\,dt
  = \underbrace{\left[\frac{\partial L}{\partial\dot{q}}\delta q\right]_{t_1}^{t_2}}_{=0\ (\text{端点固定})}
  - \int_{t_1}^{t_2}\frac{d}{dt}\!\left(\frac{\partial L}{\partial\dot{q}}\right)\delta q\,dt
  \]

  \textbf{第 4 步：利用 $\delta q$ 的任意性。}
  \[
  \delta S = \int_{t_1}^{t_2}\left[\frac{\partial L}{\partial q} - \frac{d}{dt}\!\left(\frac{\partial L}{\partial\dot{q}}\right)\right]\delta q\,dt = 0
  \]
  这对\textbf{任意} $\delta q(t)$ 成立，因此被积函数必须在每一点为零：
  \[
  \frac{d}{dt}\!\left(\frac{\partial L}{\partial\dot{q}}\right) - \frac{\partial L}{\partial q} = 0
  \]

  \textbf{物理意义}：Euler-Lagrange 方程告诉我们，系统的"惯性"（$\frac{d}{dt}\frac{\partial L}{\partial\dot{q}}$）与"驱动力"（$\frac{\partial L}{\partial q}$）在任何时刻精确平衡。这类似于 $F=ma$，但适用于任意广义坐标。
\end{derivationbox}

\section{牛顿三大定律}

\subsection{牛顿第一定律——惯性定律}

\begin{lawbox}[Newton I: 惯性定律]
\[
\mathbf{F}=0 \;\implies\; \frac{d\mathbf{v}}{dt}=0 \quad\text{（速度恒定）}
\]
\textbf{前提}：R2 + 无外力（势能 $V=$ 常数，空间均匀）

\textbf{变量}：$\mathbf{F}$ — 合外力（Newton），$\mathbf{v}$ — 速度矢量（m/s），$t$ — 时间（s）
\end{lawbox}

\begin{derivationbox}[推导]
  在无外力情况下，势能 $V(\mathbf{r})$ 是常数（空间是均匀的——没有哪个位置"更特殊"），因此 $\nabla V = 0$。
  
  自由粒子的拉格朗日量只有动能项：
  \[
  L = \frac{1}{2}m\dot{\mathbf{r}}^2
  \]
  计算 Euler-Lagrange 方程的各项：
  \[
  \frac{\partial L}{\partial \dot{\mathbf{r}}} = m\dot{\mathbf{r}} \quad\text{（动量）},\qquad
  \frac{\partial L}{\partial \mathbf{r}} = 0 \quad\text{（无外力）}
  \]
  代入得 $\frac{d}{dt}(m\dot{\mathbf{r}}) = 0$，积分得 $\dot{\mathbf{r}} = \text{常数}$。

  \textbf{物理意义}：如果空间是均匀的（没有哪个位置特殊），那么匀速运动就是自然的——没有任何理由让物体加速或减速。牛顿第一定律本质上是\textbf{空间均匀性的声明}。
\end{derivationbox}

\subsection{牛顿第二定律——动力学方程}

\begin{lawbox}[Newton II: 动力学方程]
\[
\mathbf{F} = \frac{d\mathbf{p}}{dt} = m\mathbf{a}
\]
\textbf{前提}：R2 + $L = T - V$

\textbf{变量}：
\begin{itemize}
  \item $\mathbf{F}$ — 力（Newton, N = kg·m/s$^2$），定义为势能负梯度：$\mathbf{F} \equiv -\nabla V$
  \item $\mathbf{p} = m\mathbf{v}$ — 动量（kg·m/s），经典力学中正比于速度
  \item $m$ — 惯性质量（kg），物体抵抗加速的量度
  \item $\mathbf{a} = d\mathbf{v}/dt = d^2\mathbf{r}/dt^2$ — 加速度（m/s$^2$）
\end{itemize}
\end{lawbox}

\begin{derivationbox}[推导]
  取拉格朗日量 $L = \frac{1}{2}m|\dot{\mathbf{r}}|^2 - V(\mathbf{r})$。这是经典力学最为通用的 $L$ 形式——动能 $T$ 减势能 $V$。

  计算 Euler-Lagrange 方程的每一项：
  \begin{itemize}
    \item $\displaystyle\frac{\partial L}{\partial \dot{\mathbf{r}}} = m\dot{\mathbf{r}} = \mathbf{p}$ —— 速度项给出动量
    \item $\displaystyle\frac{\partial L}{\partial \mathbf{r}} = -\nabla V(\mathbf{r})$ —— 位置项给出势能的梯度
  \end{itemize}

  代入 Euler-Lagrange 方程：
  \[
  \frac{d}{dt}(m\dot{\mathbf{r}}) = -\nabla V
  \]

  定义力为势能的负梯度：$\mathbf{F} \equiv -\nabla V$。
  
  \textbf{这并不是循环定义}——势能 $V(\mathbf{r})$ 是由物理系统的\textbf{具体配置}（如弹簧的伸长量、电荷的距离）决定的函数。一旦知道了 $V(\mathbf{r})$ 的形貌，力就由 $\mathbf{F} = -\nabla V$ 计算出。

  因此：
  \[
  \mathbf{F} = \frac{d}{dt}(m\dot{\mathbf{r}}) = m\ddot{\mathbf{r}} = m\mathbf{a}
  \]

  \textbf{物理意义}：力的概念不是原始的——它是从势能函数 $V(\mathbf{r})$ 派生出来的。不同的物理系统有不同的 $V(\mathbf{r})$：弹簧的 $V = \frac{1}{2}kx^2$，引力的 $V = -GmM/r$，静电的 $V = kq_1q_2/r$。每种力律都来自具体的势函数，而 Euler-Lagrange 方程是处理所有这些情况的统一框架。
\end{derivationbox}

\subsection{牛顿第三定律——作用与反作用}

\begin{lawbox}[Newton III: 作用力与反作用力]
\[
\mathbf{F}_{12} = -\mathbf{F}_{21}
\]
\textbf{前提}：Noether 定理（R2 推论）→ 动量守恒 → 空间平移对称性

\textbf{变量}：$\mathbf{F}_{12}$ — 物体 1 对物体 2 的作用力，$\mathbf{F}_{21}$ — 物体 2 对物体 1 的作用力
\end{lawbox}

\begin{derivationbox}[推导]
  \textbf{第 1 步：空间平移对称性产生动量守恒。}
  
  物理定律在空间平移 $\mathbf{r} \to \mathbf{r} + \boldsymbol{\epsilon}$ 下不变（北京和纽约的物理定律相同）。由 Noether 定理，这个对称性对应一个守恒量——总动量：
  \[
  \mathbf{P}_{\text{总}} = \mathbf{p}_1 + \mathbf{p}_2
  \]
  对于孤立系统：$d\mathbf{P}_{\text{总}}/dt = 0$。

  \textbf{第 2 步：动量守恒等价于作用力-反作用力关系。}
  
  将动量守恒展开：
  \[
  \frac{d\mathbf{p}_1}{dt} + \frac{d\mathbf{p}_2}{dt} = 0
  \]
  由牛顿第二定律，$\frac{d\mathbf{p}_i}{dt} = \mathbf{F}_{i\text{, 受}}$（第 $i$ 个物体受到的总力）。对于两体系统：
  \[
  \mathbf{F}_{1\text{,受}} + \mathbf{F}_{2\text{,受}} = 0
  \]
  如果两物体之间\textbf{只有}彼此施加的力（孤立系统），则 $\mathbf{F}_{1\text{,受}} = \mathbf{F}_{21}$（2 对 1 的力），$\mathbf{F}_{2\text{,受}} = \mathbf{F}_{12}$（1 对 2 的力）。因此：
  \[
  \mathbf{F}_{21} + \mathbf{F}_{12} = 0 \implies \mathbf{F}_{12} = -\mathbf{F}_{21}
  \]

  \textbf{物理意义}：牛顿第三定律不是独立的物理原则——它是动量守恒在两体系统中的表现形式。而动量守恒又来自空间的均匀性。这里的关键洞察是：\textbf{守恒律不是偶然的巧合，而是对称性的必然结果}。
\end{derivationbox}

\section{万有引力定律}

\begin{lawbox}[Newton's Law of Universal Gravitation]
\[
\mathbf{F} = -G\frac{m_1 m_2}{r^2}\,\hat{\mathbf{r}}
\]
\textbf{前提}：Einstein 场方程（R1+R2）的弱场低速极限

\textbf{变量}：
\begin{itemize}
  \item $G = 6.67430\times 10^{-11}$ m$^3$kg$^{-1}$s$^{-2}$ — 万有引力常数。所有基本常数中最不精确的（不确定度 $2.2\times 10^{-5}$）
  \item $m_1, m_2$ — 两个物体的引力质量（kg）。由等效原理（R1），引力质量精确等于惯性质量
  \item $r$ — 两物体质心之间的距离（m）
  \item $\hat{\mathbf{r}}$ — 从 $m_1$ 指向 $m_2$ 的单位向量。负号表示力是吸引力
\end{itemize}
\end{lawbox}

\begin{derivationbox}[推导：从广义相对论到牛顿引力]
  牛顿引力不是独立的物理定律，而是广义相对论在低速弱场下的近似。

  \textbf{第 1 步：Einstein 场方程的弱场极限。}
  Einstein 场方程：
  \[
  G_{\mu\nu} = \frac{8\pi G}{c^4}T_{\mu\nu}
  \]
  在弱场近似下（$|\phi| \ll c^2$），度规可写为 $g_{\mu\nu} = \eta_{\mu\nu} + h_{\mu\nu}$，其中 $|h_{\mu\nu}| \ll 1$ 是对平直时空的微小扰动。在静态（时间导数消失）和低速（$v \ll c$，应力-能量张量以 $\rho c^2$ 为主）极限下，$(0,0)$ 分量退化为 Poisson 方程：
  \[
  \nabla^2\phi = 4\pi G\rho
  \]
  其中 $\phi$ 是牛顿引力势，$\rho$ 是质量密度。

  \textbf{第 2 步：球对称解。}
  对球对称质量分布 $M$，Poisson 方程的通解为 $\phi(r) = -GM/r$（加上一个无物理意义的常数）。

  \textbf{第 3 步：检验粒子受力。}
  一个质量为 $m$ 的检验粒子在势场 $\phi$ 中的力为 $\mathbf{F} = -m\nabla\phi$。计算梯度：
  \[
  \nabla\!\left(-\frac{GM}{r}\right) = \frac{GM}{r^2}\hat{\mathbf{r}}
  \]
  因此：
  \[
  \mathbf{F} = -m\frac{GM}{r^2}\hat{\mathbf{r}} = -G\frac{mM}{r^2}\hat{\mathbf{r}}
  \]

  \textbf{物理意义}：引力的 $1/r^2$ 形式来自三维空间的几何（Gauss 定理在 $D$ 维给出 $F \propto 1/r^{D-1}$）。这也是为什么引力常数 $G$ 的量纲是 $[L^3 M^{-1} T^{-2}]$——它必须配合 $1/r^2$ 和 $m_1m_2$ 产生力的量纲。
\end{derivationbox}

\section{Kepler 行星运动定律}

\begin{lawbox}[Kepler 第三定律]
\[
T^2 = \frac{4\pi^2}{GM}\,r^3 \quad\implies\quad T^2 \propto r^3
\]
\textbf{前提}：牛顿第二定律 + 万有引力定律（圆形轨道近似）

\textbf{变量}：
\begin{itemize}
  \item $T$ — 轨道周期（s），行星绕恒星一圈的时间
  \item $r$ — 轨道半径（m），圆形轨道近似；椭圆轨道为半长轴
  \item $M$ — 中心天体的质量（kg）
  \item $G$ — 万有引力常数
\end{itemize}
\end{lawbox}

\begin{derivationbox}[推导]
  对于圆形轨道，行星的向心加速度为 $a_c = v^2/r$。

  \textbf{第 1 步：力平衡。} 引力提供向心力：
  \[
  \frac{GMm}{r^2} = m\frac{v^2}{r}
  \]

  \textbf{第 2 步：消去 $m$，解出 $v$。}
  $m$ 出现在等号两边（等效原理！），可以消去：
  \[
  \frac{GM}{r^2} = \frac{v^2}{r} \implies v^2 = \frac{GM}{r}
  \]

  \textbf{第 3 步：用轨道周长和周期表达 $v$。}
  对于半径为 $r$ 的圆形轨道，周长 $= 2\pi r$，周期 $T$ 满足：
  \[
  v = \frac{2\pi r}{T}
  \]

  \textbf{第 4 步：代入得 Kepler 第三定律。}
  \[
  \frac{4\pi^2 r^2}{T^2} = \frac{GM}{r}
  \implies T^2 = \frac{4\pi^2}{GM}r^3
  \]

  \textbf{物理意义}：比例常数 $4\pi^2/GM$ 仅依赖于中心天体的质量，与行星无关。这意味着只要测出 $T$ 和 $r$，就可以"称量"太阳的质量。这个原理推广到双星系统、系外行星探测乃至星系质量测量。
\end{derivationbox}
