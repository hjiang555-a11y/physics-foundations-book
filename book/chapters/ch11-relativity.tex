\section{Lorentz 变换}
% ════════════════════════════════════════════════════════

\begin{lawbox}[Lorentz 变换]
  $t' = \gamma(t - vx/c^2)$,\quad $x' = \gamma(x - vt)$,\quad $\gamma = 1/\sqrt{1 - v^2/c^2}$
  \quad\textbf{前提}：c 不变 (R1), Lorentz 不变性 (R1)
\end{lawbox}

\noindent\textbf{变量}：$\gamma \ge 1$ [1]（Lorentz 因子），$v$ [L$\cdot$T$^{-1}$]（相对速度，$|v| < c$）。

\begin{derivationbox}[推导]
  \textbf{Step 1} — 基本假设：$c' = c$（R1: kernel.light\_speed\_invariance）

  \textbf{Step 2} — 间隔不变性：$c^2\Delta t'^2 - \Delta x'^2 = c^2\Delta t^2 - \Delta x^2$
  \hfill\textit{(R1: kernel.lorentz\_invariance)}

  \textbf{Step 3} — 线性变换：$x' = Ax + Bt$，$t' = Cx + Dt$
  \hfill\texttt{[N]} 时空均匀 $\to$ 线性（无 $x^2$ 项）

  \textbf{Step 4} — 标准构型（S' 以速度 $v$ 相对 S 沿 $x$ 轴运动）

  \textbf{Step 5} — 代入间隔不变性，对照 $t^2$, $x^2$, $xt$ 系数：
  \begin{align*}
    t^2&: c^2D^2 - A^2v^2 = c^2\\
    x^2&: c^2C^2 - A^2 = -1\\
    xt&: 2c^2CD + 2A^2v = 0
  \end{align*}
  $\implies D = \gamma$，$A = \gamma$，$C = -\gamma v/c^2$，$B = -\gamma v$

  \textbf{Step 6} — $x' = \gamma(x-vt)$，$t' = \gamma(t-vx/c^2)$，$\gamma = 1/\sqrt{1-v^2/c^2}$
\end{derivationbox}

\texttt{reversible: true}，\texttt{[N]} $|v| < c$（$v=c$ 时 $\gamma$ 发散，$v>c$ 时 $\gamma$ 虚数——不物理）。

\section{质能等价}
% ════════════════════════════════════════════════════════

\begin{lawbox}[质能等价]
  $E^2 = p^2c^2 + m^2c^4$；静止系：$E = mc^2$
  \quad\textbf{前提}：Lorentz 不变性 (R1)
\end{lawbox}

\noindent\textbf{变量}：四维动量 $p^\mu = (E/c, \mathbf{p})$ [M$\cdot$L$\cdot$T$^{-1}$],
静止质量 $m$ [M]（Lorentz 标量——与参考系无关的不变质量），
固有时 $\tau$ [T]（$d\tau^2 = ds^2/c^2$）。

\begin{derivationbox}[推导]
  \textbf{Step 1} — 四维动量：$p^\mu \equiv m\,dx^\mu/d\tau = (\gamma mc, \gamma m\mathbf{v}) = (E/c, \mathbf{p})$
  \hfill\textit{（$m$ 是 Lorentz 标量——粒子的内禀质量不随参考系改变）}

  \textbf{Step 2} — 时间分量：$E = \gamma mc^2$

  \textbf{Step 3} — Lorentz 不变量 (R1)：$p_\mu p^\mu = (E/c)^2 - \mathbf{p}^2 = m^2c^2$

  \textbf{Step 4} — $E^2 = \mathbf{p}^2c^2 + m^2c^4$。静止系（$\mathbf{p}=0$）：$E_{\text{rest}} = mc^2$
\end{derivationbox}

\section{Einstein 场方程}
% ════════════════════════════════════════════════════════

\begin{lawbox}[Einstein 场方程]
  $G_{\mu\nu} + \Lambda g_{\mu\nu} = \dfrac{8\pi G}{c^4}\,T_{\mu\nu}$
  \quad\textbf{前提}：等效原理, 广义协变性 (R1), 最小作用量 (R2)
\end{lawbox}

\noindent\textbf{变量}：
Einstein 张量 $G_{\mu\nu} = R_{\mu\nu} - \frac{1}{2}Rg_{\mu\nu}$ [L$^{-2}$]，
Ricci 张量 $R_{\mu\nu}$ [L$^{-2}$]，
应力-能量张量 $T_{\mu\nu}$ [M$\cdot$L$^{-1}\cdot$T$^{-2}$]，
度规行列式 $g = \det(g_{\mu\nu}) < 0$ [1]。

\begin{derivationbox}[推导：Lovelock 唯一性 + Einstein-Hilbert 变分]

\noindent\textbf{Step 1} — \textbf{Lovelock 定理 (1971)}：$D$ 维时空中，由度规及其导数构成、产生至多二阶运动方程的最一般 Lagrange 密度是 \textbf{Lovelock Lagrangian}：
\begin{equation}
\mathcal{L}_{\text{grav}} = \sum_{k=0}^{\lfloor D/2\rfloor} c_k \mathcal{L}_k
\end{equation}
其中 $\mathcal{L}_k$ 为 $k$ 阶 Euler 密度：$\mathcal{L}_0=1$ ($\Lambda$)，$\mathcal{L}_1=R$ (Einstein-Hilbert)，$\mathcal{L}_2 = R^2-4R_{\mu\nu}R^{\mu\nu}+R_{\mu\nu\rho\sigma}R^{\mu\nu\rho\sigma}$ (Gauss-Bonnet)。

\noindent\textbf{Step 2} — \textbf{维度截断}：$D=4$ 时 $\lfloor D/2\rfloor=2$。但 $\mathcal{L}_2$（Gauss-Bonnet）在 4D 中为\textbf{拓扑不变量}（Chern-Gauss-Bonnet 定理），其变分恒为零，不贡献运动方程。更高阶项 $\mathcal{L}_{k\geq3}$ 在 4D 中恒为零（微分形式阶数超过流形维数）。

\noindent\textbf{Step 3} — \textbf{为什么 $R^2$ 项不合法}：$R^2$、$R_{\mu\nu}R^{\mu\nu}$ 等不是 Euler 密度——它们的变分产生 $\Box R$ 等四阶导数 $\Rightarrow$ \textbf{Ostrogradsky 鬼场}（理论不稳定，量子化后不幺正）。\fbox{[N]} 二阶场方程约束排除了这些项。

\noindent\textbf{Step 4} — \textbf{4D 中唯一的合法纯度规引力 Lagrangian}：
\begin{equation}
\mathcal{L}_{\text{grav}} = c_0 + c_1 R = -2\Lambda + \frac{R}{2\kappa}
\end{equation}
此即 Einstein-Hilbert 作用量——在 4D 中，它\textbf{不是众多选择之一，而是唯一的选择}。
$\leftrightarrow$ 交叉印证：自旋-统计定理同样依赖 $D=4$——量子统计和经典引力从两侧独立推出维度的特殊地位。

\noindent\textbf{Step 5} — Einstein-Hilbert 作用量：
$S = \frac{1}{2\kappa}\int R\sqrt{-g}\,d^4x + S_{\text{matter}}$

\noindent\textbf{Step 6} — 对 $g^{\mu\nu}$ 变分（R2：$\delta S = 0$）

\noindent\textbf{Step 7} — 引力部分变分：$\delta(\int R\sqrt{-g}\,d^4x) = \int (R_{\mu\nu} - \frac{1}{2}Rg_{\mu\nu})\delta g^{\mu\nu}\sqrt{-g}\,d^4x$
\hfill\textit{(Palatini 恒等式 + $\delta\sqrt{-g} = -\frac{1}{2}\sqrt{-g}g_{\mu\nu}\delta g^{\mu\nu}$)}

\noindent\textbf{Step 8} — 定义 $G_{\mu\nu} \equiv R_{\mu\nu} - \frac{1}{2}Rg_{\mu\nu}$。$\nabla_\mu G^{\mu\nu} \equiv 0$ (Bianchi)

\noindent\textbf{Step 9} — 物质部分：$T_{\mu\nu} \equiv -\frac{2}{\sqrt{-g}}\frac{\delta S_{\text{matter}}}{\delta g^{\mu\nu}}$

\noindent\textbf{Step 10} — 场方程：$G_{\mu\nu} = \kappa T_{\mu\nu}$。Newton 极限（弱场低速）固定 $\kappa = 8\pi G/c^4$
\end{derivationbox}

\textbf{必要条件}：R1（等效原理 [N] + 广义协变性 [N]），R2（$\delta S=0$ [N]），4 维时空 [N]（Lovelock 定理的核心——仅 $D=4$ 时唯一候选为 $\Lambda+R$），二阶场方程约束 [N]（排除 $R^2$ 等高阶鬼场项）。

\textbf{充分条件}：R1+R2+4D+二阶方程约束 $\Rightarrow$ Einstein 场方程（Lovelock 定理保证唯一性）。\texttt{reversible: true}
