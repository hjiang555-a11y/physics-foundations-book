\section{概述}

Noether 定理（1918）是 R2 的最重要推论之一。它建立了一个深刻的\textbf{双向等价}关系：作用量的每个连续对称性对应一个守恒量，每个守恒量反过来揭示一个对称性。本章从 Noether 定理出发，推导能量、动量、角动量和电荷的守恒律。

\begin{insightbox}[对称性与守恒律：一个统一的视角]
  Symmetry $\leftrightarrow$ Conservation 是物理学的黄金法则。这不是经验规律——这是 Noether \textbf{定理}，一个纯粹的数学证明。其深刻含义在于：

  \begin{itemize}
    \item \textbf{时间平移不变} $\to$ 能量守恒：今天的物理定律和昨天一样
    \item \textbf{空间平移不变} $\to$ 动量守恒：这里的物理定律和北京一样
    \item \textbf{旋转不变} $\to$ 角动量守恒：面朝东和面朝北，物理定律不变
    \item \textbf{量子相位不变}（$U(1)$）$\to$ 电荷守恒：波函数的整体相位不可观测
  \end{itemize}

  如果你发现了一个守恒量，你可以反推它背后的对称性。如果你发现一个对称性，你可以预言它必然伴随的守恒量。这个双向等价性使得 Noether 定理成为物理学中最重要的结构定理之一。
\end{insightbox}

\section{能量守恒}
% ════════════════════════════════════════════════════════

\begin{lawbox}[能量守恒]
  $dE/dt = 0$（孤立系统）；局部：$\partial_\mu T^{\mu\nu} = 0$
  \quad\textbf{前提}：Noether 定理（时间平移对称性）
\end{lawbox}

\begin{derivationbox}[推导：时间平移对称性到能量守恒]
  \textbf{Step 1} — 时间平移：$t \to t' = t + \varepsilon$，$q_i(t) \to q_i'(t) = q_i(t-\varepsilon) \approx q_i - \varepsilon\dot{q}_i$
  $\implies$ 生成元 $Q_i = -\dot{q}_i$
  \hfill\texttt{[N]} R1 的时间坐标支持平移

  \textbf{Step 2} — 若 $\partial L/\partial t = 0$（不显含时），则变换为对称性：
  $L' - L = \varepsilon\,dL/dt \implies F = -L$

  \textbf{Step 3} — Noether 荷：
  $Q = \sum_i\frac{\partial L}{\partial\dot{q}_i}(-\dot{q}_i) - (-L) = -\sum_i p_i\dot{q}_i + L = -H$

  \textbf{Step 4} — $d(-H)/dt = 0 \implies dH/dt = 0$。定义能量 $E \equiv H = \sum_i p_i\dot{q}_i - L$
  \hfill\texttt{[N]} $\partial L/\partial t = 0$——若 $L$ 显含时，$dH/dt = -\partial L/\partial t \neq 0$

  \textbf{Step 5} — $dE/dt = 0$。\quad\texttt{[S]} $\partial L/\partial t = 0$ + Noether $\Rightarrow$ 能量守恒。
\end{derivationbox}

\section{动量守恒}
% ════════════════════════════════════════════════════════

\begin{lawbox}[动量守恒]
  $d\mathbf{P}/dt = 0$（孤立系统）
  \quad\textbf{前提}：Noether 定理（空间平移对称性）
\end{lawbox}

\noindent\textbf{变量}：总动量 $\mathbf{P} = \sum_i m_i\mathbf{v}_i$ [M$\cdot$L$\cdot$T$^{-1}$]。

\begin{derivationbox}[推导：空间平移对称性到动量守恒]
  \textbf{Step 1} — 空间平移：$\mathbf{r} \to \mathbf{r} + \varepsilon\hat{\mathbf{n}}$（$\hat{\mathbf{n}}$ 为任意方向）
  $\implies$ 生成元 $Q_i = \hat{\mathbf{n}}$

  \textbf{Step 2} — 若 $V$ 仅依赖粒子间相对距离 $|\mathbf{r}_i - \mathbf{r}_j|$，则 $L$ 平移不变
  \hfill\texttt{[N]} 无外力场破坏空间均匀性

  \textbf{Step 3} — Noether 荷：
  $Q = \sum_i \frac{\partial L}{\partial\dot{\mathbf{r}}_i}\cdot\hat{\mathbf{n}} = (\sum_i m_i\mathbf{v}_i)\cdot\hat{\mathbf{n}}$

  \textbf{Step 4} — 对所有方向成立 $\implies \mathbf{P}$ 矢量守恒
  \hfill\texttt{[N]} 孤立系统（无外力）
\end{derivationbox}

\section{角动量守恒}
% ════════════════════════════════════════════════════════

\begin{lawbox}[角动量守恒]
  $d\mathbf{L}/dt = 0$（孤立系统，有心力）
  \quad\textbf{前提}：Noether 定理（旋转对称性）
\end{lawbox}

\noindent\textbf{变量}：总角动量 $\mathbf{L} = \sum_i \mathbf{r}_i \times \mathbf{p}_i$ [M$\cdot$L$^2\cdot$T$^{-1}$]，
旋转角 $\delta\theta$ [1]，旋转轴 $\hat{\mathbf{n}}$。

\begin{derivationbox}[推导：旋转对称性到角动量守恒]
  \textbf{Step 1} — 绕 $\hat{\mathbf{n}}$ 的无穷小旋转：$\mathbf{r} \to \mathbf{r} + \varepsilon(\hat{\mathbf{n}}\times\mathbf{r})$
  $\implies$ 生成元 $Q_i = \hat{\mathbf{n}} \times \mathbf{r}_i$
  \hfill\texttt{[N]} R1: $D=3 \implies$ SO(3) 旋转群

  \textbf{Step 2} — 有心力势 $V(r)$ 仅依赖 $|\mathbf{r}| \implies L$ 旋转对称
  \hfill\texttt{[N]} 势能仅依赖距离（有心力）

  \textbf{Step 3} — Noether 荷：
  $Q = \sum_i m_i\mathbf{v}_i\cdot(\hat{\mathbf{n}}\times\mathbf{r}_i) = \hat{\mathbf{n}}\cdot\sum_i(\mathbf{r}_i\times m_i\mathbf{v}_i) = \hat{\mathbf{n}}\cdot\mathbf{L}$
  \hfill\textit{(标量三重积循环置换)}

  \textbf{Step 4} — 对任意 $\hat{\mathbf{n}}$ 成立 $\implies \mathbf{L}$ 矢量守恒
\end{derivationbox}

\section{电荷守恒}
% ════════════════════════════════════════════════════════

\begin{lawbox}[电荷守恒]
  $\partial_\mu J^\mu = 0$；全局 $dQ/dt = 0$
  \quad\textbf{前提}：Noether 定理 + $U(1)$ 规范对称性 (R3)
\end{lawbox}

\noindent\textbf{变量}：守恒流 $J^\mu$（$J^0 = c\rho_q$ [L$^{-3}\cdot$T$\cdot$I]，$J^i = \mathbf{J}$ [L$^{-2}\cdot$I]），
总电荷 $Q = \int\rho_q\,d^3x$ [T$\cdot$I]，复标量场 $\psi$ [ℂ]。

\begin{derivationbox}[推导：$U(1)$ 全局对称性 $\to$ 电荷守恒]
  \textbf{Step 1} — 全局 $U(1)$ 变换（$\alpha =$ 常数）：$\psi \to e^{i\alpha}\psi$，$\psi^* \to e^{-i\alpha}\psi^*$
  \hfill\texttt{[N]} $\psi$ 场有 $U(1)$ 对称性（带电）

  \textbf{Step 2} — 复标量场 Lagrangian：$\mathcal{L} = \partial_\mu\psi^*\partial^\mu\psi - m^2\psi^*\psi$（$U(1)$ 不变）

  \textbf{Step 3} — Noether 流（场论形式）：
  $J^\mu = \frac{\partial\mathcal{L}}{\partial(\partial_\mu\psi)}(i\psi) + \frac{\partial\mathcal{L}}{\partial(\partial_\mu\psi^*)}(-i\psi^*) = i(\psi^*\partial^\mu\psi - \psi\partial^\mu\psi^*)$

  \textbf{Step 4} — $\partial_\mu J^\mu = 0$（Noether 定理直接结论）

  \textbf{Step 5} — 全局电荷守恒：
  $dQ/dt = \frac{d}{dt}\int J^0 d^3x = -\int\nabla\cdot\mathbf{J}\,d^3x = -\oint_{S_\infty}\mathbf{J}\cdot d\mathbf{A} = 0$
  \hfill\texttt{[N]} 孤立系统（无穷远处 $\mathbf{J}=0$）
\end{derivationbox}

\textbf{补充规则}：连续性方程 $\partial_\mu J^\mu = 0$ 是每个守恒律的局部微分形式（\texttt{rule.conservation\_continuity}），
它保证了守恒量的\textbf{局域守恒}——总量变化只能通过边界流实现，不能"凭空出现或消失"。

\keypoint{守恒律的局部形式 $\partial_\mu J^\mu = 0$ 告诉我们：如果你在一个小区域内数电荷（或能量、动量），任何变化都必定伴随着流入或流出的流。没有"凭空创造"或"凭空消失"。这是所有守恒律的共同数学结构——它是微分形式的"守恒"。}

\bridge{全局对称性 $\xrightarrow{\text{Noether}}$ 守恒流 $J^\mu$，$\partial_\mu J^\mu = 0$ \\ $\xrightarrow{\text{Gauss}}$ 守恒荷 $Q = \int J^0 d^3x$，$dQ/dt = 0$}{对称性 $\to$ 每一个对称变换的"方向"上存在一个不随时间改变的量 \\ 这个量在局部满足连续性方程，在全局是守恒荷——结构完全统一}

\begin{oombox}[四大守恒律的实验验证精度]
  \begin{itemize}
    \item \textbf{能量守恒}：在核反应中验证到 $\sim 10^{-7}$（$\alpha$ 粒子能量 + 反冲核能量 = 母核衰变能）
    \item \textbf{动量守恒}：粒子物理中的 Dalitz 图分析验证到 $< 10^{-3}$ 精度
    \item \textbf{角动量守恒}：天体力学（行星轨道）验证到 $10^{-8}$——Laplace 用角动量守恒解释了木星和土星轨道的稳定性
    \item \textbf{电荷守恒}：电子衰变（$e \to \nu\gamma$）未被观测到——寿命下限 $> 6.6\times 10^{28}$ 年（Borexino 实验），比宇宙年龄长 $10^{18}$ 倍。电荷守恒是检验最严格的守恒律之一
    \item \textbf{重子数守恒}（不是从 Noether 推出的！）：质子衰变未被观测到——寿命下限 $> 10^{34}$ 年（Super-Kamiokande）。重子数守恒在 SM 中是偶然的全局对称性，在 GUT 中可能破缺
  \end{itemize}
\end{oombox}

\precisionbox[四大守恒律精度]
  能量守恒：\precexact 在时间平移不变时；\precisionapprox 在宇宙学尺度（无全局时间平移对称性）\\
  动量守恒：\precexact 在空间均匀时（从 Noether 定理保证）\\
  角动量守恒：\precexact 在旋转对称时；取决于 $D=3$（SO(3) 群结构）\\
  电荷守恒：\precexact（$U(1)$ 规范对称性的精确推论）\\
  所有四个守恒律：在各自对称性精确成立时是\textbf{精确的}，在对称性近似成立时是\textbf{近似的}
\endprecisionbox

\section*{习题}

\begin{enumerate}
  \item \textbf{[简单]} 从 Noether 定理推导时间平移 $\to$ 能量守恒。哪步要求 $\partial L/\partial t=0$？若 $L$ 显含时，$dH/dt$ 等于什么？
  \item \textbf{[简单]} 两粒子系统 $L=\frac{1}{2}m_1\dot{\mathbf{r}}_1^2+\frac{1}{2}m_2\dot{\mathbf{r}}_2^2 - V(|\mathbf{r}_1-\mathbf{r}_2|)$：空间平移 $\mathbf{r}_i\to\mathbf{r}_i+\varepsilon\hat{\mathbf{n}}$ 是对称性，用 Noether 求守恒量。
  \item \textbf{[中等]} 从场论 Noether 流 $J^\mu = \frac{\partial\mathcal{L}}{\partial(\partial_\mu\phi)}\delta\phi$ 出发，对复标量场和 $U(1)$ 变换，导出电荷守恒流并验证 $\partial_\mu J^\mu=0$。
  \item \textbf{[中等]} 若 $V=kx^2+2ky^2$（破坏旋转对称），角动量的哪个分量不守恒？用 Noether 验证。
  \item \textbf{[较难]} 逆 Noether 定理：对一维系统 $H=p^2/(2m)+V(x)$，若 $H$ 守恒，能否反推 $\partial L/\partial t=0$？
  \item \textbf{[挑战]} GR 中无全局时间平移对称（膨胀宇宙），Noether 不保证全局能量守恒。但为什么太阳系仍见能量守恒？GR 中 $\nabla_\mu T^{\mu\nu}=0$ 如何退化到 $\partial_\mu T^{\mu\nu}=0$？
\end{enumerate}
